CREATE TABLE FlightType(
    FTypeId	INT PRIMARY KEY,
    FlightType VARCHAR(200) NOT NULL
);

CREATE TABLE Flight(
    FlightId INT PRIMARY KEY,
	JourneyDate DATE DEFAULT SYSDATE,
	FlightName VARCHAR(100)	NOT NULL,
	FkFlightTypeId INT NOT NULL
	FOREIGN KEY(FkFlightTypeId) REFERENCES FlightType(FTypeId)
);
